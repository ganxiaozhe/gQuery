gQuery UI 提供了丰富的布局辅助、小组件，自定义它们非常容易。

与 jQuery 一样好用的 $

  • gQuery 也同时提供了原型链及方法链，更高效且易于使用，如在原型链提供了 .wait 时延等。
  • 方法链中高度封装了许多丰富的功能，如 Storage / Cookie / Array / Date / ChainGenerator 等。
$('h2[class^=i18n-]').data({this:1, is:2, gquery:3});
[gQuery Prototype Object]
$('.is-inner > .wrap-icon').fadeOut(500).wait(1000).fadeIn();
[gQuery Dummy Object]
$.storage.set('TTL-Msg', {A:'y19',B:'boy',C:'🌈'}, '+2min');
// This will expire in 2 minutes
$.storage.get('TTL-Msg');
'{"A":"y19","B":"boy","C":"🌈"}'
$.array.finder([{role:'L'},{role:'G'},{role:'B'},{role:'T'},{role:'Q'},{role:'I'}], {role:'B'});
{
"index": 2,
"array": {
    "role": "B"
}
}
$.date(1630862585909).format('本世纪第yy年的m月d日 hh:ii:ss');
'本世纪第21年的9月6日 01:23:05'
$.date('2021-10-28').calc('+2month -30 days +56s').diff(1630862585909).ago()
'2个月前'
